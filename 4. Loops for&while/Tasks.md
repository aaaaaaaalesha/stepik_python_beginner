    x = n % 10

    if x == 3:

    if x == ld:
        cntLD += 1

    if x % 2 == 0:
        cntEven += 1

    if x > 5:
        sumGr5 += x

    if x > 7:
        multGr7 *= x

    if x == 0 or x == 5:
        cnt05 += 1
        
    n //= 10

print(cntEven)
print(sumGr5)
print(multGr7)
